import os
os.environ["PYTORCH_CUDA_ALLOC_CONF"] = "expandable_segments:True"
os.environ["TOKENIZERS_PARALLELISM"] = "false"

import random
import numpy as np
import torch
import shutil
import time
from datasets import load_dataset, Audio
from transformers import (
    WhisperFeatureExtractor, WhisperTokenizer, WhisperProcessor,
    WhisperForConditionalGeneration, Seq2SeqTrainingArguments,
    Seq2SeqTrainer
)
from peft import LoraConfig, get_peft_model

# ============================================================
# CONFIG — NOTEBOOK 1 (samples 0-48000)
# ============================================================
NOTEBOOK_ID = 1
START_SAMPLE = 0
END_SAMPLE = 48000
BASE_MODEL = "openai/whisper-large-v3"
OUTPUT_DIR = "/kaggle/working/whisper-wolof-part1"
BATCH_SIZE = 8
GRADIENT_ACCUMULATION = 2
LEARNING_RATE = 2e-4
WARMUP_STEPS = 60
MAX_STEPS = 800
EVAL_STEPS = 800
SAVE_STEPS = 400
LOGGING_STEPS = 10
AUGMENT_PROB = 0.3
MAX_TRAIN_SECONDS = 3.5 * 3600  # 3h30 — arrêt pour sauvegarder avant 4h

os.makedirs(OUTPUT_DIR, exist_ok=True)
start_time = time.time()

print(f"\n{'='*60}")
print(f"   KAGGLE NOTEBOOK {NOTEBOOK_ID}/3 — WOLOF ASR")
print(f"   Samples: {START_SAMPLE:,} a {END_SAMPLE:,}")
print(f"   ~{(END_SAMPLE-START_SAMPLE)*7/3600:.0f}h d'audio wolof")
print(f"   Max steps: {MAX_STEPS} | Duree max: 4h (arret a 3h30)")
print(f"{'='*60}")

n_gpus = torch.cuda.device_count()
print(f"\nGPUs: {n_gpus}")
for i in range(n_gpus):
    print(f"  GPU {i}: {torch.cuda.get_device_name(i)} ({torch.cuda.get_device_properties(i).total_memory / 1e9:.1f} GB)")

# ============================================================
# DATASET
# ============================================================
print(f"\nTelechargement samples {START_SAMPLE:,} a {END_SAMPLE:,}...")
train_ds = load_dataset("soynade-research/Wolof-ASR-Data", split=f"train[{START_SAMPLE}:{END_SAMPLE}]")
test_ds = load_dataset("soynade-research/Wolof-ASR-Data", split="test[:200]")
print(f"Charge: {len(train_ds):,} train, {len(test_ds):,} test")

cols = train_ds.column_names
text_col = next((c for c in ["sentence", "text", "transcription"] if c in cols), None)
audio_col = next((c for c in ["audio", "path", "audio_path"] if c in cols), None)
if text_col and text_col != "sentence":
    train_ds = train_ds.rename_column(text_col, "sentence")
    test_ds = test_ds.rename_column(text_col, "sentence")
if audio_col and audio_col != "audio":
    train_ds = train_ds.rename_column(audio_col, "audio")
    test_ds = test_ds.rename_column(audio_col, "audio")

keep = ["audio", "sentence"]
train_ds = train_ds.remove_columns([c for c in train_ds.column_names if c not in keep])
test_ds = test_ds.remove_columns([c for c in test_ds.column_names if c not in keep])
train_ds = train_ds.filter(lambda x: x["sentence"] and len(x["sentence"].strip()) > 0)
test_ds = test_ds.filter(lambda x: x["sentence"] and len(x["sentence"].strip()) > 0)

train_ds = train_ds.cast_column("audio", Audio(sampling_rate=16000))
test_ds = test_ds.cast_column("audio", Audio(sampling_rate=16000))
print(f"Apres filtrage: {len(train_ds):,} train, {len(test_ds):,} test")

elapsed = time.time() - start_time
print(f"Temps ecoule: {elapsed/60:.1f} min")

# ============================================================
# PROCESSOR
# ============================================================
feature_extractor = WhisperFeatureExtractor.from_pretrained(BASE_MODEL)
tokenizer = WhisperTokenizer.from_pretrained(BASE_MODEL, language="french", task="transcribe")
processor = WhisperProcessor.from_pretrained(BASE_MODEL, language="french", task="transcribe")

MAX_LABEL_LENGTH = 448
def filter_labels(example):
    return len(tokenizer(example["sentence"]).input_ids) < (MAX_LABEL_LENGTH - 5)
train_ds = train_ds.filter(filter_labels, num_proc=2)
test_ds = test_ds.filter(filter_labels, num_proc=2)
print(f"Post-filtre: {len(train_ds):,} train, {len(test_ds):,} test")

# ============================================================
# AUGMENTATION + DATASET CLASS
# ============================================================
def augment_audio(audio_array, sr=16000):
    audio = audio_array.copy().astype(np.float32)
    if random.random() < 0.4:
        audio += np.random.randn(len(audio)).astype(np.float32) * random.uniform(0.002, 0.015)
    if random.random() < 0.3:
        speed = random.uniform(0.9, 1.1)
        indices = np.round(np.arange(0, len(audio), speed)).astype(int)
        audio = audio[indices[indices < len(audio)]]
    if random.random() < 0.4:
        audio *= random.uniform(0.6, 1.4)
    mx = np.max(np.abs(audio))
    if mx > 0:
        audio = audio / mx * 0.95
    return audio.astype(np.float32)

class WhisperDataset(torch.utils.data.Dataset):
    def __init__(self, hf_dataset, feature_extractor, tokenizer, augment=False):
        self.dataset = hf_dataset
        self.feature_extractor = feature_extractor
        self.tokenizer = tokenizer
        self.augment = augment
    def __len__(self):
        return len(self.dataset)
    def __getitem__(self, idx):
        item = self.dataset[idx]
        audio_array = np.array(item["audio"]["array"], dtype=np.float32)
        sr = item["audio"]["sampling_rate"]
        if self.augment and random.random() < AUGMENT_PROB:
            audio_array = augment_audio(audio_array, sr)
        input_features = self.feature_extractor(audio_array, sampling_rate=sr).input_features[0]
        labels = self.tokenizer(item["sentence"]).input_ids
        return {"input_features": input_features, "labels": labels}

train_dataset = WhisperDataset(train_ds, feature_extractor, tokenizer, augment=True)
test_dataset = WhisperDataset(test_ds, feature_extractor, tokenizer, augment=False)
print(f"Dataset pret: {len(train_dataset):,} train, {len(test_dataset):,} test")

# ============================================================
# MODELE FP16 + LoRA (sans bitsandbytes)
# ============================================================
print("\nChargement Whisper Large V3 (fp16)...")
model = WhisperForConditionalGeneration.from_pretrained(
    BASE_MODEL, torch_dtype=torch.float16, device_map="auto"
)
model.config.use_cache = False
model.generation_config.language = "fr"
model.generation_config.task = "transcribe"
model.generation_config.forced_decoder_ids = None

model.enable_input_require_grads()
lora_config = LoraConfig(
    r=32, lora_alpha=64,
    target_modules=["q_proj", "v_proj", "k_proj", "out_proj", "fc1", "fc2"],
    lora_dropout=0.05, bias="none",
)
model = get_peft_model(model, lora_config)
print("Parametres entrainables:")
model.print_trainable_parameters()

elapsed = time.time() - start_time
print(f"\nTemps ecoule: {elapsed/60:.1f} min")

# ============================================================
# COLLATOR + METRICS
# ============================================================
@dataclass
class DataCollatorSpeechSeq2SeqWithPadding:
    processor: Any
    decoder_start_token_id: int
    def __call__(self, features):
        input_features = [{"input_features": f["input_features"]} for f in features]
        batch = self.processor.feature_extractor.pad(input_features, return_tensors="pt")
        label_features = [{"input_ids": f["labels"]} for f in features]
        labels_batch = self.processor.tokenizer.pad(label_features, return_tensors="pt")
        labels = labels_batch["input_ids"].masked_fill(labels_batch.attention_mask.ne(1), -100)
        if (labels[:, 0] == self.decoder_start_token_id).all().cpu().item():
            labels = labels[:, 1:]
        batch["labels"] = labels
        return batch

data_collator = DataCollatorSpeechSeq2SeqWithPadding(
    processor=processor,
    decoder_start_token_id=processor.tokenizer.convert_tokens_to_ids("<|startoftranscript|>"),
)
metric = evaluate.load("wer")

def compute_metrics(pred):
    pred_ids = pred.predictions
    label_ids = pred.label_ids
    label_ids[label_ids == -100] = tokenizer.pad_token_id
    pred_str = tokenizer.batch_decode(pred_ids, skip_special_tokens=True)
    label_str = tokenizer.batch_decode(label_ids, skip_special_tokens=True)
    return {"wer": 100 * metric.compute(predictions=pred_str, references=label_str)}

# ============================================================
# ENTRAINEMENT
# ============================================================
training_args = Seq2SeqTrainingArguments(
    output_dir=OUTPUT_DIR,
    per_device_train_batch_size=BATCH_SIZE // max(1, n_gpus),
    per_device_eval_batch_size=BATCH_SIZE // max(1, n_gpus),
    gradient_accumulation_steps=GRADIENT_ACCUMULATION,
    learning_rate=LEARNING_RATE,
    lr_scheduler_type="cosine",
    warmup_steps=WARMUP_STEPS,
    weight_decay=0.01,
    max_steps=MAX_STEPS,
    gradient_checkpointing=True,
    fp16=True,
    eval_strategy="steps",
    eval_steps=EVAL_STEPS,
    save_steps=SAVE_STEPS,
    save_total_limit=2,
    logging_steps=LOGGING_STEPS,
    report_to=["tensorboard"],
    load_best_model_at_end=False,
    predict_with_generate=True,
    generation_max_length=225,
    remove_unused_columns=False,
    label_names=["labels"],
    push_to_hub=False,
    dataloader_num_workers=2,
)

from transformers import TrainerCallback

class TimeoutCallback(TrainerCallback):
    """Arrete l'entrainement avant le timeout de 4h."""
    def on_step_end(self, args, state, control, **kwargs):
        elapsed = time.time() - start_time
        if elapsed > MAX_TRAIN_SECONDS:
            print(f"\n TIMEOUT: {elapsed/3600:.1f}h ecoulees - arret propre pour sauvegarde")
            control.should_training_stop = True

trainer = Seq2SeqTrainer(
    args=training_args, model=model,
    train_dataset=train_dataset, eval_dataset=test_dataset,
    data_collator=data_collator, compute_metrics=compute_metrics,
    processing_class=processor.feature_extractor,
    callbacks=[TimeoutCallback()],
)

print(f"\n{'='*60}")
print(f"   LANCEMENT — NOTEBOOK {NOTEBOOK_ID}")
print(f"   {MAX_STEPS} steps | Batch effectif: {BATCH_SIZE * GRADIENT_ACCUMULATION}")
print(f"   ARRET AUTO a 3h30 pour sauvegarder avant 4h")
print(f"{'='*60}\n")

trainer.train()

# ============================================================
# EVALUATION + SAUVEGARDE
# ============================================================
print("\nEvaluation...")
results = trainer.evaluate()

print(f"   NOTEBOOK {NOTEBOOK_ID} TERMINE !")

print(f"   Duree: {elapsed/60:.1f} min")

print(f"{'='*60}")
